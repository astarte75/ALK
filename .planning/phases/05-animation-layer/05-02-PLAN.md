  test('video element has autoplay attributes', async ({ page }) => {
    await page.goto('/')

    await expect(video).toBeVisible()

    await expect(video).toHaveAttribute('muted', '')

    await expect(video).toHaveAttribute('loop', '')

  test('headline renders with word spans for split-text animation', async ({ page }) => {

  test('scroll indicator is present', async ({ page }) => {

test.describe('Homepage sections (HOME-03 to HOME-07)', () => {

  })

  test('stat counters render with data attributes (HOME-06)', async ({ page }) => {
    await page.goto('/')
    const stats = page.locator('[data-stat-value]')
    const count = await stats.count()
    expect(count).toBeGreaterThanOrEqual(4)
  })

  test('news preview shows cards (HOME-05)', async ({ page }) => {
    await page.goto('/')
    const cards = page.locator('[data-news-card]')
    const count = await cards.count()
    expect(count).toBeGreaterThanOrEqual(1)
    expect(count).toBeLessThanOrEqual(3)
  })

  test('newsletter strip renders with email input (HOME-07)', async ({ page }) => {
    await page.goto('/')
    const input = page.locator('input[type="email"]')
    await expect(input).toBeVisible()
  })

  test('page loads without JS errors (HOME-08 Lenis)', async ({ page }) => {
    const errors: string[] = []
    page.on('console', (msg) => {
      if (msg.type() === 'error') errors.push(msg.text())
    })
    await page.goto('/')
    await page.waitForLoadState('networkidle')
    // Filter out known non-critical errors (e.g., failed video fetch in test env)
    const criticalErrors = errors.filter(
      (e) => !e.includes('ERR_FILE_NOT_FOUND') && !e.includes('net::')
    )
    expect(criticalErrors).toHaveLength(0)
  })
})
