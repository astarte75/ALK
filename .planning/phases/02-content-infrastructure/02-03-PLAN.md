   ```ts
   import 'dotenv/config'
   // CRITICAL: Import actual typed fetchers to prove runtime behavior (SC2)
   import {
     getPortfolioCompanies,
     getPortfolioCompanyBySlug,
     getTeamMembers,
     getFunds,
     getNewsArticles,
     getInvestmentPlatforms,
     getPageBySlug,
     getSiteConfig,
     getOffices,
   } from '../src/lib/contentful/fetchers'

   async function validate() {
     let allPassed = true

     // Test typed fetchers with IT locale
     const companies = await getPortfolioCompanies('it')
     check('Portfolio Companies (IT)', companies.length >= 15)

     const companiesEN = await getPortfolioCompanies('en')
     check('Portfolio Companies (EN)', companiesEN.length >= 15)

     const team = await getTeamMembers('it')
     check('Team Members', team.length >= 15)

     const funds = await getFunds('it')
     check('Funds', funds.length >= 5)

     const platforms = await getInvestmentPlatforms('it')
     check('Investment Platforms', platforms.length >= 3)

     const news = await getNewsArticles('it')
     check('News Articles', news.length >= 5)

     const config = await getSiteConfig('it')
     check('Site Config', config !== null)

     const offices = await getOffices('it')
     check('Offices', offices.length >= 2)

     // Test slug-based fetch (proves typed single-entity fetcher)
     if (companies.length > 0) {
       const slug = companies[0].fields.slug
       const bySlug = await getPortfolioCompanyBySlug(slug, 'it')
       check('Slug fetch (portfolioCompany)', bySlug !== null)
     }

     // Test locale mapping (proves 'it' → 'it-IT' and 'en' → 'en-US' work)
     if (companies.length > 0 && companiesEN.length > 0) {
       check('Locale mapping', companies[0].fields.name !== undefined && companiesEN[0].fields.name !== undefined)
     }

     // Test TypeScript types at runtime (field access without errors)
     if (team.length > 0) {
       const member = team[0]
       check('TypeScript type safety (team)', typeof member.fields.name === 'string' && typeof member.fields.slug === 'string')
     }

     if (allPassed) {
       console.log('\n✓ All content validation checks passed!')
       process.exit(0)
     } else {
       console.log('\n✗ Some checks failed.')
       process.exit(1)
     }

     function check(label: string, ok: boolean) {
       console.log(`${ok ? '✓' : '✗'} ${label}`)
       if (!ok) allPassed = false
     }
   }

   validate()
   ```

   NOTE: This script imports the project's actual typed fetchers, proving that:
   - `getPortfolioCompanies('it')` returns `PortfolioCompany[]` at runtime (SC2)
   - Locale mapping works (`'it'` → `'it-IT'`)
   - Slug-based fetching works
   - TypeScript types match actual Contentful field structure

4. Create `tests/revalidate.spec.ts` — Playwright tests for webhook route:
   ```ts
   import { test, expect } from '@playwright/test'

   test.describe('ISR Revalidation Webhook', () => {
     const baseUrl = 'http://localhost:3000'
     const validSecret = process.env.CONTENTFUL_WEBHOOK_SECRET || 'alkemia-webhook-secret-change-me'

     test('POST with valid secret returns 200 and revalidated:true', async ({ request }) => {
       const response = await request.post(`${baseUrl}/api/revalidate`, {
         headers: {
           'x-contentful-webhook-secret': validSecret,
           'content-type': 'application/json',
         },
         data: {
           sys: {
             type: 'Entry',
             contentType: { sys: { id: 'portfolioCompany' } },
           },
         },
       })
       expect(response.status()).toBe(200)
       const body = await response.json()
       expect(body.revalidated).toBe(true)
       expect(body.contentType).toBe('portfolioCompany')
     })

     test('POST with invalid secret returns 401', async ({ request }) => {
       const response = await request.post(`${baseUrl}/api/revalidate`, {
         headers: {
           'x-contentful-webhook-secret': 'wrong-secret',
           'content-type': 'application/json',
         },
         data: { sys: { type: 'Entry' } },
       })
       expect(response.status()).toBe(401)
     })

     test('POST with Asset type revalidates contentful-asset tag', async ({ request }) => {
       const response = await request.post(`${baseUrl}/api/revalidate`, {
         headers: {
           'x-contentful-webhook-secret': validSecret,
           'content-type': 'application/json',
         },
         data: {
           sys: { type: 'Asset' },
         },
       })
       expect(response.status()).toBe(200)
       const body = await response.json()
       expect(body.revalidated).toBe(true)
       expect(body.tag).toBe('contentful-asset')
     })
   })
   ```
